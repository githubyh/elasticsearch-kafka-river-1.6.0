001package com.silicon.plugin.kafka.listener;
002
003import org.elasticsearch.action.ActionListener;
004import org.elasticsearch.action.update.UpdateResponse;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.ESLoggerFactory;
007
008import com.silicon.plugin.kafka.Util.Utils;
009
010public class UpdateActionListener implements ActionListener<UpdateResponse> {
011    private final static ESLogger logger = ESLoggerFactory.getLogger(UpdateActionListener.class.getName());
012
013        private String message;
014        public UpdateActionListener(String message){
015                this.message = message;
016        }
017        
018        @Override
019        public void onFailure(Throwable e) {
020                logger.warn("## update action failed for message[{}]" + Utils.LINE_SEPARATOR, e, message);
021        }
022
023        @Override
024        public void onResponse(UpdateResponse response) {
025                //nothing
026        }
027
028}
